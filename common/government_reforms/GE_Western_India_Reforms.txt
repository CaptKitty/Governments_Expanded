# T1
uidapur_state_reform = {
	icon = uidapur_state_reform_icon
	allow_normal_conversion = yes
	lock_level_when_selected = yes
	potential = {
		OR = {
			have_had_reform = uidapur_state_reform
			OR = {
				is_or_was_tag = { tag = MER }
				is_or_was_tag = { tag = MAW }
			}
		}
	}
	trigger = {
		OR = {
			have_had_reform = uidapur_state_reform
			OR = {
				is_or_was_tag = { tag = MER }
				is_or_was_tag = { tag = MAW }
			}
		}
	}
	valid_for_nation_designer = yes
	nation_designer_cost = 20
	modifiers = {
		army_tradition = 0.25
		inflation_reduction = 0.10
		fort_maintenance_modifier = -0.10
		monarch_military_power = 1
		tolerance_own = 1

		allowed_rajput_fraction = 0.10
		rajput_loyalty_modifier = 0.10
		rajput_influence_modifier = 0.10
		can_recruit_rajputs = yes
	}
	government_abilities = {
		rajput_princedom_mechanic
	}
	effect = {
		hidden_effect = {
			ge_uidapur_state_update = yes
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				remove_province_modifier = ge_mer_city_of_lakes
				remove_province_modifier = ge_maw_blue_city
			}
		}
	}
	custom_attributes = {
		enables_uidapur_state = yes
	}
}

		}
	}
	trigger = {
		OR = {
			have_had_reform = krsi_trivate_reform
			culture_group = dravidian
			capital_scope = {
				OR = {
					region = deccan_region
					region = coromandel_region
				}
			}
		}
	}
	valid_for_nation_designer = yes
	nation_designer_cost = 20
	modifiers = {
		manpower_recovery_speed = 0.10
		global_autonomy = -0.025
	}
	custom_attributes = {
		enable_cistern = yes #-15% local development cost in Arid Provinces
	}
	effect = {
		hidden_effect = {
			capital_scope = {
				area = {
					every_neighbor_province = {
						limit = { owned_by = ROOT }
						add_province_modifier = {
							name = ge_cistern_modifier
							duration = -1
						}
					}
				}
			}
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				remove_province_modifier = ge_cistern_modifier
			}
		}
	}
}
